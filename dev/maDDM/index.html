model = maaDDM()

mutable struct Transition state::Int n::Int mat::Array{Float64,2} end

function Transition(mat) n = size(mat,1) state = rand(1:n) return Transition(state, n, mat) end

function attend(transition) (;mat,n,state) = transition w = mat[state,:] nextstate = sample(1:n, Weights(w)) transition.state = nextstate return next_state end
